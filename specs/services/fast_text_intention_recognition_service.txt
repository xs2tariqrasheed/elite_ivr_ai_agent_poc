create fast text intention recognition service.

It should expect a text input and return the intention label and confidence score.

Intention can be "new_reservation" or "other".

use "fasttext_reservation_training.txt" as training data (include common STT duplicates, e.g. repeated menu phrases with commas).

use "intent_model.bin" as locally running model file.


e.g.

# fast_text_script.py
# ---------------------------------------------
# Train and use a fastText model for intent classification
# Intents: new_reservation, other
# ---------------------------------------------

import fasttext
import os
import time


TRAIN_FILE = "fasttext_reservation_training.txt"
MODEL_FILE = "intent_model.bin"


def train_model():
    """Train fastText supervised model"""

    model = fasttext.train_supervised(
        input=TRAIN_FILE, epoch=25, lr=1.0, wordNgrams=2, verbose=2, minCount=1
    )

    model.save_model(MODEL_FILE)
    print(f"Model saved to {MODEL_FILE}")

    return model


def load_model():
    """Load trained model"""
    if not os.path.exists(MODEL_FILE):
        raise FileNotFoundError("Model not found. Train first.")
    return fasttext.load_model(MODEL_FILE)


def predict_intent(model, text):
    """Predict intent from text"""

    labels, probabilities = model.predict(text)

    intent = labels[0].replace("__label__", "")
    confidence = probabilities[0]

    return intent, confidence
